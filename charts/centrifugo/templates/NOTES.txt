{{- $namespace := .Release.Namespace }}

=== ACCESSING CENTRIFUGO ===

{{- if .Values.ingress.enabled }}

External access via Ingress:
{{- range $host := .Values.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}{{ . }}
  {{- end }}
{{- end }}

{{- else if contains "NodePort" .Values.service.type }}

External access via NodePort:

  export NODE_PORT=$(kubectl get --namespace {{ $namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ $fullName }})
  export NODE_IP=$(kubectl get nodes --namespace {{ $namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo "Centrifugo URL: http://$NODE_IP:$NODE_PORT"

{{- else if contains "LoadBalancer" .Values.service.type }}

External access via LoadBalancer:

  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
  Watch the status with:
    kubectl get --namespace {{ $namespace }} svc -w {{ $fullName }}

  export SERVICE_IP=$(kubectl get svc --namespace {{ $namespace }} {{ $fullName }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo "Centrifugo URL: http://$SERVICE_IP:{{ .Values.service.port }}"

{{- else if contains "ClusterIP" .Values.service.type }}

External port (client connections):

  kubectl --namespace {{ $namespace }} port-forward svc/{{ $fullName }} {{ .Values.service.port }}:{{ .Values.service.port }}

  Then access: http://localhost:{{ .Values.service.port }}

{{- end }}

=== INTERNAL ENDPOINTS ===

Internal port provides: API, Prometheus metrics, admin UI, health checks.

  kubectl --namespace {{ $namespace }} port-forward svc/{{ $fullName }} {{ .Values.serviceInternal.port }}:{{ .Values.serviceInternal.port }}

  - Admin UI:    http://localhost:{{ .Values.serviceInternal.port }}/admin
  - Metrics:     http://localhost:{{ .Values.serviceInternal.port }}/metrics
  - Health:      http://localhost:{{ .Values.serviceInternal.port }}/health
